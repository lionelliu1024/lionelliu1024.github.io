-- nb_transmit_conf: table
CREATE TABLE `nb_transmit_conf`
(
    `id`        int(11)                          NOT NULL AUTO_INCREMENT,
    `NB_COM_ID` int(11)                          NOT NULL COMMENT '运营平台id',
    `protocol`  varchar(50) COLLATE utf8mb4_bin  NOT NULL COMMENT '数据转换协议 TransmitLocalDeploy:本地部署 UnifyPushDataNB:对外推送 YiChangXiaoFang:宜昌消防 ',
    `url`       varchar(255) COLLATE utf8mb4_bin NOT NULL COMMENT '推送地址',
    `remark`    varchar(255) COLLATE utf8mb4_bin DEFAULT NULL,
    PRIMARY KEY (`id`),
    UNIQUE KEY `unq_push_url` (`NB_COM_ID`, `protocol`, `url`)
) ENGINE = InnoDB
  AUTO_INCREMENT = 9
  DEFAULT CHARSET = utf8mb4
  COLLATE = utf8mb4_bin COMMENT ='nb数据转发表';
